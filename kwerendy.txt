SELECT opony.nr_kat, opony.producent, opony.model, opony.sezon, opony.cena FROM opony ORDER BY cena ASC LIMIT 10;

SELECT opony.producent, opony.model, opony.sezon, opony.cena FROM opony WHERE opony.nr_kat = 9;

SELECT zamowienie.id_zam, zamowienie.ilosc, opony.model, opony.cena FROM zamowienie INNER JOIN opony ON opony.nr_kat=zamowienie.nr_kat ORDER BY RAND() LIMIT 1;

UPDATE opony SET cena=cena * 0.75 WHERE sezon = "letnia";
